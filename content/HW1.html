HW1-1: 將左右碰撞移動的紅色方塊, 改為先移動到水平畫面中心點位置後, 接著上下碰撞移動.

from browser import html
from browser import document as doc
import browser.timer

canvas = html.CANVAS(width = 300, height = 300)
canvas.id = "game-board3"
brython_div = doc["brython_div3"]
brython_div <= canvas
ctx = canvas.getContext("2d")

px = 0
py = 150
width = 20
height = 20
speed = 2
def game():
    global px, py, width, height, speed
    ctx.clearRect(px, py, width, height)
    ctx.fillStyle = "red"
    if px < (canvas.width/2):
        px += speed
    else:
        py += speed
    
        if py < 0 or (py + height) > canvas.height:
            speed = -speed
            
    ctx.fillRect(px, py, width, height)
    
browser.timer.set_interval(game, 10)

HW1-2: 加入與 Cango 頁面 Fourbar 程式碼相同功能的啟動按鈕, 讓使用者可以利用按鈕啟動或中斷紅色方塊上下碰撞移動. 完成後請將程式碼存入個人 Gist 區.

from browser import html
from browser import document as doc
from browser import timer
import browser.timer

canvas = html.CANVAS(width = 400, height = 100)
canvas.id = "game-board3"
brython_div = doc["brython_div3"]
brython_div <= canvas
brython_div <= html.BUTTON("開始", id="start")
ctx = canvas.getContext("2d")
px = 0
py = 50
width = 20
height = 20
speedH = 2
speedV = 2

def squre():
    global px, py, width, height, speedH, speedV
    ctx.clearRect(px, py, width, height)
    ctx.fillStyle = "red"
    px += speedH
    if px > 200:
        speedH=0
        py += speedV
    if py < 0 or (py + height) > canvas.height:
        speedV = -speedV
    ctx.fillRect(px, py, width, height)
    
squre()
    
game = None
    
def gametion(ev):
    global game
    if game is None :
        game = timer.set_interval(squre, 10)
        doc['start'].text = '暫停'
    elif game == 'hold':
        game = timer.set_interval(squre, 10)
        doc['start'].text = '暫停'
    else:
        timer.clear_interval(game)
        game = 'hold'
        doc['start'].text = '繼續'
doc["start"].bind("click", gametion)
